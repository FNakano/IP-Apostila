<pre data-mduml-processed="true">int *ordenaPorInsercaoArray(int *array, int comprimentoDoArray, int indiceDoPrimeiroForaDaOrdem) {
  if (indiceDoPrimeiroForaDaOrdem&gt;=comprimentoDoArray)         // c44*1
    return array;    // condição de parada: array ordenado   // c45*1
  int key=array[indiceDoPrimeiroForaDaOrdem];                  // c46*1
  int i=indiceDoPrimeiroForaDaOrdem-1;                         // c47*1
  while ((i&gt;=0) &amp;&amp; (array[i]&gt;key)) {             // c48*t_{I}
    array[i+1]=array[i];                                     // c49*t_{I}-1
    i--;                                                     // c50*t_{I}-1
  }
  array[i+1]=key;                                              // c52*1
  ordenaPorInsercaoArray (array, comprimentoDoArray, indiceDoPrimeiroForaDaOrdem+1);
  return array;                                                // c54
}
